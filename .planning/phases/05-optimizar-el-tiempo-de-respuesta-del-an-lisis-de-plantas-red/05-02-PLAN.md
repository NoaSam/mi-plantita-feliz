  ```typescript
  console.log("[identify] step 4: fetch edge function (SSE)");
  const response = await fetch(EDGE_URL, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "apikey": ANON_KEY,
      "Authorization": `Bearer ${session?.access_token ?? ANON_KEY}`,
    },
    body: JSON.stringify(requestBody),
    signal: AbortSignal.timeout(INVOKE_TIMEOUT_MS),
  });

  if (!response.ok) {
    const body = await response.json().catch(() => ({}));
    throw new Error(body?.error || `HTTP ${response.status}`);
  }

  const reader = response.body!.getReader();
  const decoder = new TextDecoder();
  let sseBuffer = "";
  let plantSearchId: string | null = null;
  let createdAt: string | null = null;

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    sseBuffer += decoder.decode(value, { stream: true });
    const events = sseBuffer.split("\n\n");
    sseBuffer = events.pop() ?? "";

    for (const raw of events) {
      const eventName = raw.match(/^event: (.+)$/m)?.[1];
      const dataLine = raw.match(/^data: (.+)$/m)?.[1];
      if (!dataLine) continue;
      const payload = JSON.parse(dataLine);

      if (eventName === "result") {
        const plantResult: PlantResult = {
          id: "",  // will be updated when "done" event arrives
          name: payload.name,
          description: payload.description,
          care: payload.care,
          diagnosis: payload.diagnosis,
          imageUrl: compressed,
          date: new Date().toISOString(),
          model: payload.model,
        };
        console.log("[identify] step 5: setResult (from SSE result event)");
        setResult(plantResult);

        track("plant_identified", {
          plant_name: payload.name,
          logged_in: loggedIn,
          winning_model: payload.model,
          has_location: !!coords,
        });
      }

      if (eventName === "error") {
        throw new Error(payload.error);
      }

      if (eventName === "done") {
        plantSearchId = payload.plant_search_id;
        createdAt = payload.created_at;
        // Update result with DB id and created_at
        setResult((prev) => prev ? {
          ...prev,
          id: payload.plant_search_id ?? prev.id,
          date: payload.created_at ?? prev.date,
        } : prev);
      }
    }
  }

  // localStorage for anonymous history (after done event provides the id)
  if (!session?.user && plantSearchId) {
    try {
      const currentResult = { /* rebuild from last known state */ };
      // Use a callback-based approach to read current result state
      setResult((prev) => {
        if (prev) {
          const history = JSON.parse(localStorage.getItem("plant-history") || "[]");
          history.unshift({ ...prev, id: plantSearchId });
          localStorage.setItem("plant-history", JSON.stringify(history.slice(0, 20)));
        }
        return prev;
      });
    } catch {
      // localStorage full or unavailable
    }
  }
  ```

  IMPORTANT: The `setResult` for the `done` event uses the functional updater form `setResult((prev) => ...)` because it needs to update the existing result object, not replace it. Similarly, the localStorage write uses `setResult` callback to read the current state.

  The `withTimeout` wrapper is no longer needed for the fetch call because `AbortSignal.timeout(INVOKE_TIMEOUT_MS)` handles it natively. Remove the `withTimeout` function entirely.
